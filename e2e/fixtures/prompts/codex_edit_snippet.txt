// Sample code snippet used as realistic diff content for codex e2e test payloads

package service

import (
    "crypto/hmac"
    "crypto/sha256"
    "encoding/hex"
    "fmt"
)

func ComputeRecordSig(secret, tokenKey, deviceID, timestamp, tool, filePath, repoURL, currentSHA string, added, removed int64, diffHunk *string) string {
    hunkStr := ""
    if diffHunk != nil {
        hunkStr = *diffHunk
    }
    h := sha256.Sum256([]byte(hunkStr))
    diffHash := hex.EncodeToString(h[:])
    canonical := fmt.Sprintf("%s\n%s\n%s\n%s\n%s\n%s\n%s\n%d\n%d\n%s",
        tokenKey, deviceID, timestamp, tool, filePath, repoURL, currentSHA, added, removed, diffHash)
    mac := hmac.New(sha256.New, []byte(secret))
    mac.Write([]byte(canonical))
    return hex.EncodeToString(mac.Sum(nil))
}
